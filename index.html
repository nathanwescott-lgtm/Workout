// ===== Countdown Logic =====

function updateCountdown() {

  const holiday = new Date("October 25, 2025 00:00:00").getTime();

  const now = new Date().getTime();

  const distance = holiday - now;


  if (distance < 0) {

    document.getElementById("countdown").innerText = "Holiday time! 🏝";

    return;

  }

  const days = Math.floor(distance / (1000 * 60 * 60 * 24));

  document.getElementById("countdown").innerText =

    days + " days until holiday – stay on track!";

}

setInterval(updateCountdown, 1000 * 60 * 60);

updateCountdown();

// ===== Weight Tracking =====

let weightHistory = JSON.parse(localStorage.getItem("weightHistory")) || [];


function saveWeight() {

  const weight = parseFloat(document.getElementById("weight-input").value);

  if (!isNaN(weight)) {

    const date = new Date().toLocaleDateString();

    weightHistory.push({ date, weight });

    localStorage.setItem("weightHistory", JSON.stringify(weightHistory));

    renderWeightHistory();

  }

}


function renderWeightHistory() {

  const list = weightHistory.map(entry => `<li>${entry.date}: ${entry.weight} kg</li>`).join("");

  document.getElementById("weight-history").innerHTML = `<ul>${list}</ul>`;

}


renderWeightHistory();
